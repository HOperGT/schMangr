<script>
        function copySchedule() {
            const selectedDay = document.getElementById("template-day").value;
            const modifiableLessons = document.querySelectorAll(".modifiable-lesson");
        
            // Копирование расписания из выбранного дня в изменяемое расписание
            modifiableLessons.forEach((lesson, index) => {
                lesson.value = document.querySelector(`input[name="${selectedDay}"]`).value.split(',')[index] || '';
            });
        }
        
        function sendSchedule() {
            const form = document.getElementById('schedule-form');
            const formData = new FormData(form);
            const classId = window.location.pathname.split('/').pop(); // Получаем id класса из адресной строки
        
            // Отправка данных пользователям через AJAX
            fetch(`/send_schedule/${classId}`, {
                method: 'POST',
                body: formData,
            })
            .then(response => response.json())
            .then(data => {
                if (data.success) {
                    alert('Расписание успешно отправлено всем пользователям!');
                } else {
                    alert('Ошибка при отправке расписания.');
                }
            })
            .catch(error => {
                console.error('Ошибка:', error);
            });
        }
    </script>
